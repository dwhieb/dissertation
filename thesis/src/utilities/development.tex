% COMMANDS

% \addcite{}
%
% Add a placeholder citation
%
% Example:
%
% Is is the subject of much debate whether these language-specific categories can be mapped onto each other or compared in any useful way \addcite{Haspelmath 2010, other publications on this debate}.
\newcommand{\addcite}[1]{%
  \hl{(CITE: #1)}%
  \todo[
    caption = {citation},
    color   = magenta,
    noline,
    size    = \tiny
  ]{}%
}

% \addcrossref{}
%
% Add a cross-reference
%
% Example:
%
% For a complete list of languages used in this thesis, see the \addcrossref{List of Languages}.
\newcommand{\addcrossref}[1]{%
  \textcolor{blue}{#1}%
  \todo[
    caption = {cross reference},
    color   = blue,
    noline,
    size    = \tiny
  ]{}%
}

% \TODO{}
%
% Add a todo item
%
% Example:
%
% \TODO[caption = {This is the caption.}]{This is a todo item.}
\newcommand{\TODO}[2][]{%
  \todo[caption = {#2}, #1]{\begin{spacing}{1}#2\end{spacing}}%
}
