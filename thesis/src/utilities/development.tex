% COMMANDS

% \addcite{}
%
% Add a placeholder citation
%
% Example:
%
% Is is the subject of much debate whether these language-specific categories can be mapped onto each other or compared in any useful way \addcite{Haspelmath 2010, other publications on this debate}.
%
\newcommand{\addcite}[1]{%
  \hl{(CITE: #1)}%
  \todo[
    caption={citation},
    color=magenta,
    noline,
    size=\tiny
  ]{}%
}

% \addissue{}{}
%
% Add a todo note that references a GitHub issue
%
% Example:
%
% \todo{add Abstract}{357}
%
\newcommand{\addissue}[2]{%
  \todo{#1 (\href{https://github.com/dwhieb/dissertation/issues/#2}{\##2})}%
}

% \addref{}
%
% Add a cross-reference
%
% Example:
%
% For a complete list of languages used in this thesis, see the \addref{List of Languages}.
%
\newcommand{\addref}[1]{%
  \textcolor{blue}{#1}%
  \todo[
    caption={cross reference},
    color=blue,
    noline,
    size=\tiny
  ]{}%
}
